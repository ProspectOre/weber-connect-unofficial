#include <tunables/global>

# AppArmor profile for the unofficial Weber Connect for Home Assistant add-on.
# The profile name must match the add-on slug so Supervisor loads it.
profile weber_connect_ble flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/python>
  #include <abstractions/dbus-strict>

  # Receive the graceful-shutdown signals Supervisor sends on stop.
  signal (receive) set=(kill,term,int,hup,cont),

  # S6-Overlay init system from the Home Assistant base image.
  # S6's /init launcher is interpreted by /bin/sh on current HA base images,
  # so AppArmor must permit the interpreter to read it as well as execute it.
  /init rix,
  /run.sh rix,
  # S6 launches several interpreted helper scripts from these paths. AppArmor
  # therefore needs read permission alongside inherited execution permission.
  /bin/** rix,
  /usr/bin/** rix,
  /usr/local/bin/** rix,
  /run/{s6,s6-rc*,service}/** rix,
  /package/** rix,
  /command/** rix,
  /etc/s6-overlay/ r,
  /etc/s6-overlay/** rix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # bashio + the add-on options provided by Supervisor.
  /usr/lib/bashio/** rix,
  /data/options.json r,

  # libc resolver configuration used by curl, bashio, and the Python cloud
  # client. Keep this read-only and limited to name-service configuration.
  /etc/host.conf r,
  /etc/hosts r,
  /etc/nsswitch.conf r,
  /etc/resolv.conf r,

  # TLS trust store used by Python's HTTPS client for the opt-in Weber cloud
  # fallback. Certificates are public and remain read-only.
  /etc/ssl/cert.pem r,
  /etc/ssl/certs/ r,
  /etc/ssl/certs/** r,

  # Python runtime and the add-on application code (read/execute only).
  /usr/bin/python3* ix,
  /usr/lib/python3*/** r,
  /usr/local/lib/python3*/** r,
  /app/ r,
  /app/** r,

  # Persistent add-on state (pairing summary, settings, status snapshots).
  /data/ rw,
  /data/** rwk,
  /tmp/ rw,
  /tmp/** rwk,

  # System D-Bus for the BlueZ BLE backend (Bleak + bluetoothctl).
  dbus send bus=system path=/org/bluez/**
       interface=org.bluez.*,
  dbus send bus=system path=/org/bluez/**
       interface=org.freedesktop.DBus.*,
  dbus receive bus=system path=/org/bluez/**,
  dbus (send,receive) bus=system path=/
       interface=org.freedesktop.DBus.ObjectManager,
  dbus send bus=system path=/org/freedesktop/DBus
       interface=org.freedesktop.DBus,
  /usr/bin/bluetoothctl ix,

  # The ingress HTTP panel, MQTT, and opt-in HTTPS cloud fallback use TCP.
  # DNS resolution for Supervisor and cloud hostnames requires UDP datagrams;
  # raw sockets and network-administration capabilities remain forbidden.
  network unix stream,
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  # Alpine/musl consults interface and route metadata through NETLINK_ROUTE
  # while resolving names. This does not permit raw IP packet sockets.
  network netlink raw,
}
